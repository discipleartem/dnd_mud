```python
    """–†–µ–∞–ª–∏–∑–∞—Ü–∏—è —Ä–µ–ø–æ–∑–∏—Ç–æ—Ä–∏—è —Å SQLAlchemy."""

        """–°–æ—Ö—Ä–∞–Ω–∏—Ç—å –ø–µ—Ä—Å–æ–Ω–∞–∂–∞ –≤ –±–∞–∑—É –¥–∞–Ω–Ω—ã—Ö."""

            name=character.name,
            level=character.level,
            hit_points=character.hit_points
        )

        self.session.commit()
        
        character.id = db_character.id
        return character
    
    def find_by_id(self, character_id: int) -> Character:

            CharacterModel.id == character_id
        ).first()
        
        if not db_character:
            return None
        
        return Character(
            id=db_character.id,
            name=db_character.name,
            level=db_character.level,
            hit_points=db_character.hit_points
        )
```
